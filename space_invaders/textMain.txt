#include <SFML/Graphics.hpp>
#include <iostream>
#include <chrono>
#include <array>
#include <random>

using namespace sf;
//using namespace std;

#include <cmath>

int main() {
	sf::Event event;

	sf::RenderWindow window(VideoMode(720, 720), "Space Invaders", Style::Close);
	window.setView(View(FloatRect(0, 0, 720, 720)));
	while (1 == window.isOpen())
	{
		while (1 == window.pollEvent(event))
		{
			if (Event::Closed)
			{
				window.close();
			}
		}
		window.clear();
		window.display();
	}
	 
	return 0;
}
